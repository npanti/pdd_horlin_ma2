1function index=timeSynchronisation(input)
 2
 3global d
 4x=zeros(1,d.subCarriers); %utilise pour calculer la correlation
 5indexMax=length(input)-2*length(x);
 6correlation=zeros(1,indexMax);
 7
 8%Compute cross-correlation

10    Ps=signalPower(input(i:d.subCarriers*d.numberPreamble+i-1));

12    correlation(i)=abs(x*input(i+d.subCarriers:2*d.subCarriers+i-1));

18correlationMean=zeros(1,indexMax-d.cyclicPrefixPreamble);
19for j=1:indexMax-d.cyclicPrefixPreamble
20    correlationMean(j)=mean(correlation(j:d.cyclicPrefixPreamble+j-1));
21end
22
23[~,index]=max(correlationMean);

26if(d.synchronisationPlots==1)
