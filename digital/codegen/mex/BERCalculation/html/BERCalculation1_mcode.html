1function BER=BERCalculation(bitsRX,type)
 2
 3%type=0 ->preamble
 4%type=1 ->data
 5
 6global d
 7
 8if(type==1)
 9    incorrectBits=0;
10    for m=1:length(bitsRX)
11        if(bitsRX(m)~=d.data(m))
12            incorrectBits=incorrectBits+1;
13        end
14    end
15    BER=incorrectBits/d.messageLength;
16    fprintf('BER for data: %.10f\n',BER);
17    %disp(['BER for data: ',num2str(BER)]);
18elseif(type==0)
19    incorrectBits=0;
20    for m=1:length(bitsRX)
21        if(bitsRX(m)~=d.preamble(m))
22            incorrectBits=incorrectBits+1;
23        end
24    end
25    BER=incorrectBits/length(bitsRX);
26    fprintf('BER for preamble: %.10f\n',BER);
27    %disp(['BER for preamble: ',num2str(BER)]);
28else
29    BER = -1;
30end
31
32end
